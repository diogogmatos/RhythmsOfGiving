CREATE TABLE Licitador (
    id INT PRIMARY KEY,
    nome VARCHAR(45),
    palavraPasse VARCHAR(45),
    dataNascimento DATE,
    nrCartao INT,
    email VARCHAR(100),
    nif INT,
    numeroCC BIGINT,
    UNIQUE (nif, email, numeroCC, id)
);

CREATE TABLE Licitacao (
    id INT PRIMARY KEY UNIQUE,
    dataHora DATATIME,
    valor DECIMAL(7, 2),
    idLicitador INT,
    idLeilao INT
);

CREATE TABLE Leilao (
    id INT PRIMARY KEY UNIQUE,
    titulo VARCHAR(45),
    valorAtual DECIMAL(7, 2),
    valorMinimo DECIMAL(7, 2),
    dataHoraFinal DATATIME,
    localizacao VARCHAR(100),
    ativo BIT ,
    tipoLeilao VARCHAR(45),
    descricao VARCHAR(MAX),
    imagem VARBINARY(MAX),
    idArtista INT,
    idGeneroMusical INT,
    idAdministrador INT
);

CREATE TABLE Administrador (
    id INT PRIMARY KEY,
    palavraPasse VARCHAR(45),
    email VARCHAR(100),
    UNIQUE (id, email)
);

CREATE TABLE Instituicao (
    id INT PRIMARY KEY,
    nome VARCHAR(45),
    descricao VARCHAR(MAX),
    logotipo VARBINARY(MAX),
    hiperligacao VARCHAR(100),
    idAdministrador INT,
    UNIQUE (id, nome)
);

CREATE TABLE Artista (
    id INT PRIMARY KEY,
    imagem VARBINARY(MAX),
    nome VARCHAR(45),
    idAdministrador INT,
    UNIQUE (id, nome)
);

CREATE TABLE Fatura (
    id INT PRIMARY KEY UNIQUE,
    dataHora DATATIME,
    idLicitador INT,
    idLicitacao INT,
    idInstituicao INT
);

CREATE TABLE GeneroMusical (
    id INT PRIMARY KEY,
    nome VARCHAR(45),
    idAdministrador INT,
    UNIQUE (nome, id)
);

CREATE TABLE Notificação (
    idNotificacao INT PRIMARY KEY,
    mensagem VARCHAR(100),
    titulo VARCHAR(45),
    dataHora DATATIME,
    tipo INT,
    idLicitador INT
);
